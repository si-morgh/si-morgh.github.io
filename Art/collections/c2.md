<div class="carousel-overlay" id="galleryOverlay">
  <button class="close-gallery" id="closeBtn" aria-label="Close Gallery">&times;</button>

  <div class="carousel-container">
    <div class="carousel-slide" id="carouselSlide">
      {% for img in page.images %}
        <div class="art-item">
          <img src="{{ '/assets/images/art/collections/C1/' | append: img.file | relative_url }}" alt="{{ img.title }}">
          <div class="art-caption">{{ img.title }}</div>
        </div>
      {% endfor %}
    </div>

    <div id="image-counter">1 / {{ page.images.size }}</div>
    <button class="carousel-prev" id="prevBtn" aria-label="Previous">❮</button>
    <button class="carousel-next" id="nextBtn" aria-label="Next">❯</button>
  </div>
</div>

<script>
  const slide = document.getElementById('carouselSlide');
  const nextBtn = document.getElementById('nextBtn');
  const prevBtn = document.getElementById('prevBtn');
  const counterDisplay = document.getElementById('image-counter');
  const items = document.querySelectorAll('.art-item');

  let counter = 0;
  const total = items.length;

  function updateSlide() {
    // This moves the "film strip" horizontally
    slide.style.transform = `translateX(${-100 * counter}%)`;
    // Updates the counter text (e.g., 2 / 5)
    counterDisplay.innerText = `${counter + 1} / ${total}`;
  }

  nextBtn.addEventListener('click', () => {
    counter = (counter >= total - 1) ? 0 : counter + 1;
    updateSlide();
  });

  prevBtn.addEventListener('click', () => {
    counter = (counter <= 0) ? total - 1 : counter - 1;
    updateSlide();
  });

  // Keyboard support for that "pro" feel
  document.addEventListener('keydown', (e) => {
    if (e.key === "ArrowLeft") prevBtn.click();
    if (e.key === "ArrowRight") nextBtn.click();
  });
  const closeBtn = document.getElementById('closeBtn');
const overlay = document.getElementById('galleryOverlay');

closeBtn.addEventListener('click', () => {
  overlay.style.display = 'none'; // Simply hides the show
});

// Also allow exiting by pressing the "Escape" key
document.addEventListener('keydown', (e) => {
  if (e.key === "Escape") {
    overlay.style.display = 'none';
  }
});
</script>
